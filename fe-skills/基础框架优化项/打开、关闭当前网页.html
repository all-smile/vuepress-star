export const isPc = () => {
  const ua = navigator.userAgent
  const moveDevices = [
    "Android",
    "iPhone",
    "SymbianOS",
    "Windows Phone",
    "iPad",
    "iPod"
  ]
  let flag = true;
  for (let i = 0; i < moveDevices.length; i++) {
    if (ua.indexOf(moveDevices[i]) !== -1) {
      flag = false;
      break
    }
  }
  return flag
}
